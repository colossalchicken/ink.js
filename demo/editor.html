//set BG color
document.documentElement.style.background = 'white'

car = {
    width: 200,
    height: 100,
    rotate: -90,
    x: window.innerWidth / 4 - 100,
    y: window.innerHeight - 200,
    type: 'image',
    source: 'car.png',
}

car2 = {
    width: 200,
    height: 100,
    rotate: 90,
    x: window.innerWidth / 2 - 100,
    y: -200,
    type: 'image',
    source: 'car.png',
}

car3 = {
    width: 200,
    height: 100,
    rotate: 90,
    x: window.innerWidth / 4 - 100,
    y: -200,
    type: 'image',
    source: 'car.png',
}

scoreboard = {
    x: 20,
    y: 10,
    contents: "move your mouse to play",
}

mouse.cursor = 'none'

//increases over time
score = 0

forever = () =>
{
    if (car2.y < page.height + 200)
    {
        car2.y += 10 * score * .01
    }
    else
    {
        car2.y = -100
        car2.x = random(0, page.width/2 - 300)
    }
    if (car3.y < page.height + 200)
    {
        car3.y += 10 * score * .02
    }
    else
    {
        car3.y = -100
        car3.x = random(100, page.width/2 - 300)
    }
    if (score != 0)
    {
        score += .05
    }
    if (score != 0)
    {
        scoreboard.contents = 'score: ' + (Math.round(score * 100) - 10000)
    }
    if (car.x > car2.x - 100 && car.x < car2.x + 100 && car2.y > page.height - 400 && car2.y < page.height - 200)
    {
        score = 100
        forever = () =>
        {}
        mouse.moved = () =>
        {}
        mouse.cursor = ''
        scoreboard.contents = 'score: ' + (Math.round(score * 100) - 10000) + "<h1> You crashed!</h1><br><button onclick='window.location.reload()'>restart</button</h1>"
    }
    if (car.x > car3.x - 100 && car.x < car3.x + 100 && car3.y > page.height - 400 && car3.y < page.height - 200)
    {
        score = 100
        forever = () =>
        {}
        mouse.moved = () =>
        {}
        mouse.cursor = ''
        scoreboard.contents = 'score: ' + (Math.round(score * 100) - 10000) + "<h1> You crashed!</h1><br><button onclick='window.location.reload()'>restart</button</h1>"
    }
}

mouse.moved = () =>
{
    setTimeout(() =>
    {
        car.x = mouse.x - 100
    }, 10);
    if (score == 0)
    {
        score = 100
    }
}
